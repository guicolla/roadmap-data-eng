# Funcao principal de transformacao def transform(details: list[dict]) -> pd.DataFrame: records = [] for d in details: stats = {s["stat"]["name"]: s["base_stat"] for s in d["stats"]} records.append({ "nome": d["name"].capitalize(), "tipo_primario": d["types"][0]["type"]["name"], "hp": stats.get("hp", 0), ... # demais campos }) df = pd.DataFrame(records) # Converter unidades df["altura_m"] = df["altura_dm"] / 10 df["peso_kg"] = df["peso_hg"] / 10 # Classificar por poder df["categoria"] = pd.cut( df["total_stats"], bins=[0, 300, 400, 500, 600, 999], labels=["Fraco", "Mediano", "Forte", "Muito Forte", "Lendario"] ) return df
